// Firebase Storage Security Rules
// Copy and paste these rules into Firebase Console > Storage > Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Allow read/write access to all files for development
    // ⚠️ WARNING: This is for development only! Update these rules for production.
    match /{allPaths=**} {
      allow read, write: if true;
    }
  }
}
